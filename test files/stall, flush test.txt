//stall, flush test

module inst_mem(
input [29:0] pc,
output reg [31:0] inst);

always @(*) begin

    case (pc)

    //================================================
    // Initialize Registers
    //================================================

    0  : inst = 32'hAAAAD0B7; // lui  x1,0xAAAAD
    1  : inst = 32'hCCC08093; // addi x1,x1,-820
                               // x1 = AAAACCCC

    2  : inst = 32'h55555137; // lui  x2,0x55555
    3  : inst = 32'h55510113; // addi x2,x2,1365
                               // x2 = 55555555
    4  : inst = 32'h00400193; //x3 =4
    5  : inst = 32'h00302023; // at 0x0 = 4
    //================================================
    // LOAD-USE HAZARD (STALL TEST)
    //================================================

    6  : inst = 32'h00002183; // lw x3,0(x0)

    // immediate dependency on x3
    // should cause ONE STALL

    7  : inst = 32'h00118213; // addi x4,x3,1

    8  : inst = 32'h004202B3; // add x5,x4,x4


    //================================================
    // BRANCH NOT TAKEN
    //================================================

    9  : inst = 32'h00028463; // beq x5,x0,+8
                               // NOT taken

    10  : inst = 32'h00100313; // addi x6,x0,1
    11  : inst = 32'h00200393; // addi x7,x0,2


    //================================================
    // BRANCH TAKEN (FLUSH TEST)
    //================================================

    12 : inst = 32'h00000663; // beq x0,x0,+12
                               // ALWAYS taken

    // these instructions should be FLUSHED

    13 : inst = 32'h11100413; // addi x8,x0,0x111
    14 : inst = 32'h22200493; // addi x9,x0,0x222

    // branch target

    15 : inst = 32'h33300513; // addi x10,x0,0x333
    
    //jump test
    
    16 : inst = 32'h00c005EF; // jal x11,+8

    // should be flushed

    17 : inst = 32'h44400613; // addi x12,x0,0x444
    18 : inst = 32'h55500693; // addi x13,x0,0x555

    // jump target

    19 : inst = 32'h66600713; // addi x14,x0,0x666


    default : inst = 32'h00000013; // nop

    endcase

end
endmodule
